  <Steps>
    <Step title="Clone the repository">
      Clone the Cinemusic repository to your local machine:
      
      ```bash
      git clone https://github.com/JuanPabloLeonF/cinemusic.git
      cd cinemusic
      ```
    </Step>
    
    <Step title="Install dependencies">
      Install all required npm packages:
      
      ```bash
      npm install
      ```
    </Step>
    
    <Step title="Start development server">
      Launch the Angular development server:
      
      ```bash
      ng serve
      ```
      
      Open your browser and navigate to `http://localhost:4200/` to see the application running.
    </Step>
  </Steps>
